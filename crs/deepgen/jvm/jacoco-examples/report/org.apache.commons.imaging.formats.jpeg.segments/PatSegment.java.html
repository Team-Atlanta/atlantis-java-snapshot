package org.apache.commons.imaging.formats.jpeg.segments;

import java.io.IOException;
import java.io.InputStream;
import java.nio.charset.StandardCharsets;
import java.util.Arrays;

import javax.xml.parsers.DocumentBuilderFactory;
import javax.xml.parsers.DocumentBuilder;
import javax.xml.xpath.XPath;
import javax.xml.xpath.XPathFactory;
import javax.xml.xpath.XPathExpression;
import javax.xml.xpath.XPathConstants;
import org.w3c.dom.Document;
import org.w3c.dom.NodeList;
import org.xml.sax.ErrorHandler;
import org.xml.sax.SAXException;
import org.xml.sax.SAXParseException;


public class PatSegment extends GenericSegment {
    private final String key;
    private String entryPoint = null;

    public PatSegment(final int marker, final byte[] bytes) {
        super(marker, bytes);

        key = getKeyFromSegmentData(getSegmentData());
        if (key.length() + 1 < getSegmentData().length) {
            entryPoint = getEntryPointFromSegmentData(key, Arrays.copyOfRange(getSegmentData(), key.length() + 1, getSegmentData().length));
        }
    }

    public PatSegment(final int marker, final int markerLength, final InputStream is) throws IOException {
        super(marker, markerLength, is);

        key = getKeyFromSegmentData(getSegmentData());
        if (key.length() + 1 < getSegmentData().length) {
            entryPoint = getEntryPointFromSegmentData(key, Arrays.copyOfRange(getSegmentData(), key.length() + 1, getSegmentData().length));
        }
    }

    private static String getKeyFromSegmentData(byte[] segmentData) {
        // Return the content of the segment until the first 0x00 byte
        for (int i = 0; i < segmentData.length; i++) {
            if (segmentData[i] == 0x00) {
                return new String(Arrays.copyOf(segmentData, i), StandardCharsets.UTF_8);
            }
        }
        return new String(segmentData, StandardCharsets.UTF_8);
    }

    private static String getEntryPointFromSegmentData(String key, byte[] xmlData) {
        // Use XPath to get the element with the key
        try {
            DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
            DocumentBuilder builder = factory.newDocumentBuilder();

            builder.setErrorHandler(new ErrorHandler() {
                @Override
                public void warning(SAXParseException exception) throws SAXException {
                    // Do nothing
                }
                @Override
                public void error(SAXParseException exception) throws SAXException {
                    // Do nothing
                }
                @Override
                public void fatalError(SAXParseException exception) throws SAXException {
                    // Do nothing
                }
            });

            Document doc = builder.parse(new java.io.ByteArrayInputStream(xmlData));
            XPathFactory xPathfactory = XPathFactory.newInstance();
            XPath xpath = xPathfactory.newXPath();
            XPathExpression expr = xpath.compile("//entry[@key='" + key + "']");
            NodeList nl = (NodeList) expr.evaluate(doc, XPathConstants.NODESET);
            if (nl.getLength() > 0) {
                return nl.item(0).getTextContent();
            }
        } catch (Exception e) {
            // Ignore
        }

        return null;
    }

    /**
     * Returns a copy of the path.
     *
     * @return a copy of the path's bytes
     */
    public byte[] getPath() {
        return getSegmentData();
    }

    @Override
    public String getDescription() {
        return "PAT (" + getSegmentDataAsString(StandardCharsets.UTF_8) + ")";
    }

}
